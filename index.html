customPopup.js

customPopup.js는 간편하게 팝업을 생성할 수 있는 Javascript 플러그인 입니다.
별도의 CSS 없이 팝업을 생성할 수 있으며 일정기간 열지 않기 기능을 제공합니다.

  • 사용방법

    HTML

    
  <script src="/js/customPopup.js"></script> <!-- import customPopup.js -->

  <!-- 팝업 마크업 생성 -->
  <div id="popup"></div> <!-- ※ 쿠키가 해당 마크업의 id값으로 생성되기 때문에 고유의 id 값을 가진 태그를 생성해주어야 함! -->

    Javascript

    
  $('#popup').customPopup({
    width: '300px', // 팝업 사이즈 [기본값 300px]
    top: '50px', // 위치값 [기본값 0px]
    left: '50px', // 위치값 [기본값 0px]
    imgSrc: 'https://via.placeholder.com/600x700/000000/ffffff?text=Pop-up-img', // 이미지 경로
    imgAlt: '첫 번째 팝업 이미지', // 이미지 alt 속성
    period: 1, // 다시 열지 않기 기간(1 = 1일) [기본값 1]
    borderColor: '#000' // 팝업 및 버튼 테두리 color [기본값 #999999] 
  });
